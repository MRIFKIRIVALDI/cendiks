8ï¸âƒ£ Contoh Lengkap: Sistem Manajemen Buku

Aplikasi perpustakaan mini dengan module! ğŸ“šâœ¨

# File: buku.py
class Buku:
    def __init__(self, judul, penulis):
        self.judul = judul
        self.penulis = penulis
        self.tersedia = True

# File: perpustakaan.py
from buku import Buku

class Perpustakaan:
    def __init__(self):
        self.buku = []
    
    def tambah_buku(self, judul, penulis):
        self.buku.append(Buku(judul, penulis))
    
    def tampilkan_buku(self):
        for b in self.buku:
            print(f"{b.judul} - {b.penulis}")

# File: main.py
from perpustakaan import Perpustakaan

lib = Perpustakaan()
lib.tambah_buku("Python", "Guido")
lib.tambah_buku("Java", "James")
lib.tampilkan_buku()
