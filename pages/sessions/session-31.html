2Ô∏è‚É£ Mengimport Modul

Berbagai cara mengimport modul.

# Cara 1: Import lengkap
import math
print(math.sqrt(16))  # Output: 4.0

# Cara 2: Import dengan alias
import math as m
print(m.pi)  # Output: 3.141592653589793

# Cara 3: Import spesifik
from math import sqrt, pi
print(sqrt(25))  # Output: 5.0
print(pi)        # Output: 3.141592653589793

# Cara 4: Import semua (tidak direkomendasikan)
from math import *
print(sin(0))    # Output: 0.0

3Ô∏è‚É£ Membuat Modul Sendiri

Cara membuat dan menggunakan modul kustom.

# Buat file kalkulator.py
def tambah(a, b):
    return a + b

def kurang(a, b):
    return a - b

def kali(a, b):
    return a * b

def bagi(a, b):
    if b != 0:
        return a / b
    return "Error: Division by zero"

# Gunakan modul
import kalkulator

print(kalkulator.tambah(10, 5))  # Output: 15
print(kalkulator.kurang(10, 5))  # Output: 5
print(kalkulator.kali(10, 5))    # Output: 50
print(kalkulator.bagi(10, 5))    # Output: 2.0

4Ô∏è‚É£ Modul Built-in Python

Modul yang sudah tersedia dalam Python.

# Modul math
import math
print(math.sqrt(16))    # Output: 4.0
print(math.pi)          # Output: 3.141592653589793
print(math.sin(math.pi/2))  # Output: 1.0

# Modul random

# Modul datetime
import datetime

print(sekarang)  # Output: 2023-12-07 10:30:45.123456
print(sekarang.year)   # Output: 2023
print(sekarang.month)  # Output: 12

5Ô∏è‚É£ Apa Itu Package?

Direktori yang berisi modul-modul dan file __init__.py.

# Struktur package:
# mypackage/
#   __init__.py
#   module1.py
#   module2.py
#   subpackage/
#     __init__.py
#     module3.py

# Menggunakan package
import mypackage.module1
mypackage.module1.function1()

# Atau
from mypackage import module1
module1.function1()

# Menggunakan subpackage
from mypackage.subpackage import module3
module3.function3()

6Ô∏è‚É£ Membuat Package

Cara membuat package sederhana.

# Buat struktur folder:
# utilities/
#   __init__.py
#   math_utils.py
#   string_utils.py

# Isi __init__.py
# from .math_utils import *
# from .string_utils import *

# Isi math_utils.py
def factorial(n):
    if n == 0:
        return 1
    return n * factorial(n-1)

# Isi string_utils.py
def reverse_string(text):
    return text[::-1]

# Menggunakan package
from utilities import factorial, reverse_string

print(factorial(5))        # Output: 120
print(reverse_string("hello"))  # Output: olleh

7Ô∏è‚É£ Modul vs Package

Perbedaan antara modul dan package.

# Modul: File Python tunggal (.py)
# Contoh: math.py, random.py

# Package: Direktori yang berisi modul-modul
# Contoh: numpy/, pandas/, mypackage/

# Package harus memiliki __init__.py
# Modul tidak perlu file khusus

# Penggunaan:
import math          # Modul
import numpy as np   # Package

from math import sqrt     # Dari modul
from numpy import array   # Dari package

8Ô∏è‚É£ Contoh Penggunaan

Program yang menggunakan modul dan package.

import math
import random
from datetime import datetime

def generate_random_data():
    """Generate data acak untuk testing"""
    data = []
    for _ in range(10):
        # Generate angka random antara 1-100
        num = random.randint(1, 100)
        # Hitung akar kuadrat
        sqrt_num = math.sqrt(num)
        # Bulatkan ke 2 desimal
        rounded = round(sqrt_num, 2)
        data.append(rounded)
    return data

def main():
    print("Program Demo Modul dan Package")
    print("=" * 30)

    # Generate data
    data = generate_random_data()
    print(f"Data yang dihasilkan: {data}")

    # Hitung statistik sederhana
    rata_rata = sum(data) / len(data)
    maksimum = max(data)
    minimum = min(data)

    print(f"Rata-rata: {rata_rata:.2f}")
    print(f"Nilai maksimum: {maksimum}")
    print(f"Nilai minimum: {minimum}")

    # Timestamp
    waktu_sekarang = datetime.now()
    print(f"Waktu eksekusi: {waktu_sekarang.strftime('%Y-%m-%d %H:%M:%S')}")

if __name__ == "__main__":
    main()
