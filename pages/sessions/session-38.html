1Ô∏è‚É£ Apa Itu Regular Expression?

Pola untuk mencocokkan dan memanipulasi teks.

import re

# Contoh sederhana: mencari kata "Python" dalam teks
teks = "Saya sedang belajar Python programming"
pola = r"Python"

hasil = re.search(pola, teks)
if hasil:
    print(f"Ditemukan '{hasil.group()}' pada posisi {hasil.start()}-{hasil.end()}")
else:
    print("Tidak ditemukan")

# Menggunakan findall untuk menemukan semua kemunculan
teks2 = "Python adalah bahasa programming. Python sangat powerful."
hasil_semua = re.findall(pola, teks2)
print(f"Jumlah kemunculan: {len(hasil_semua)}")
print(f"Hasil: {hasil_semua}")

# Menggunakan match (hanya di awal string)
hasil_match = re.match(r"Saya", teks)
if hasil_match:
    print("String dimulai dengan 'Saya'")
else:
    print("String tidak dimulai dengan 'Saya'")

2Ô∏è‚É£ Karakter Literal dan Meta

Dasar-dasar pola regex.

import re

teks = "Halo dunia! Saya belajar Python123."

# Karakter literal (cocok persis)
print("1. Karakter literal:")
print(re.findall(r"Python", teks))  # ['Python']

# Meta karakter . (dot) - cocok dengan karakter apapun
print("\n2. Meta karakter dot:")
print(re.findall(r"P.th.n", teks))  # ['Python']

# Meta karakter ^ (caret) - awal string
print("\n3. Meta karakter caret:")
print(re.findall(r"^Halo", teks))  # ['Halo']

# Meta karakter $ (dollar) - akhir string
print("\n4. Meta karakter dollar:")
print(re.findall(r"Python123.$", teks))  # ['Python123.']

# Meta karakter * (asterisk) - 0 atau lebih kemunculan
print("\n5. Meta karakter asterisk:")
print(re.findall(r"Pyth.*", teks))  # ['Python123.']

# Meta karakter + (plus) - 1 atau lebih kemunculan
print("\n6. Meta karakter plus:")
print(re.findall(r"Pyth.+", teks))  # ['Python123.']

# Meta karakter ? (question) - 0 atau 1 kemunculan
print("\n7. Meta karakter question:")
print(re.findall(r"Python\d?", teks))  # ['Python1'] (karena diikuti angka)

# Meta karakter \d - digit (0-9)
print("\n8. Meta karakter \\d:")
print(re.findall(r"\d+", teks))  # ['123']

# Meta karakter \w - word character (a-z, A-Z, 0-9, _)
print("\n9. Meta karakter \\w:")
print(re.findall(r"\w+", teks))  # ['Halo', 'dunia', 'Saya', 'belajar', 'Python123']

# Meta karakter \s - whitespace
print("\n10. Meta karakter \\s:")
print(re.findall(r"\s+", teks))  # [' ', ' ', ' ']

3Ô∏è‚É£ Character Classes

Mencocokkan karakter dari set tertentu.

import re

teks = "Email: user@example.com, Phone: 08123456789, Date: 25-12-2023"

# Character class [abc] - salah satu dari a, b, atau c
print("1. Character class [abc]:")
print(re.findall(r"[aeiou]", teks))  # ['a', 'e', 'a', 'e', 'o', 'e']

# Character class [a-z] - range huruf kecil
print("\n2. Character class [a-z]:")
print(re.findall(r"[a-z]+", teks))  # ['mail', 'user', 'example', 'com', 'hone', 'ate']

# Character class [A-Z] - range huruf besar
print("\n3. Character class [A-Z]:")
print(re.findall(r"[A-Z][a-z]+", teks))  # ['Email', 'Phone', 'Date']

# Character class [0-9] - digit
print("\n4. Character class [0-9]:")
print(re.findall(r"[0-9]+", teks))  # ['08123456789', '25', '12', '2023']

# Character class [^abc] - negasi (bukan a, b, atau c)
print("\n5. Character class [^abc]:")
print(re.findall(r"[^aeiou\s:,-]+", teks))  # karakter non-vokal, non-spasi, dll

# Predefined character classes
print("\n6. Predefined classes:")
print("Digits:", re.findall(r"\d+", teks))      # ['08123456789', '25', '12', '2023']
print("Words:", re.findall(r"\w+", teks))       # ['Email', 'user', 'example', 'com', 'Phone', '08123456789', 'Date', '25', '12', '2023']
print("Spaces:", re.findall(r"\s+", teks))      # [' ', ' ', ' ', ' ', ' ', ' ']
print("Non-digits:", re.findall(r"\D+", teks))  # ['Email: user@example.com, Phone: ', ', Date: ', '-','-']

# Contoh validasi email sederhana
def is_valid_email(email):
    pattern = r"^[\w\.-]+@[\w\.-]+\.\w+$"
    return bool(re.match(pattern, email))

print("\n7. Validasi email:")
emails = ["user@example.com", "invalid-email", "user.name@domain.co.uk"]
for email in emails:
    print(f"{email}: {'Valid' if is_valid_email(email) else 'Invalid'}")

4Ô∏è‚É£ Quantifiers

Menentukan jumlah kemunculan karakter.

import re

teks = "aa ab aab aaab aaaab aaaaab"

# * - 0 atau lebih
print("1. Quantifier * (0 atau lebih):")
print(re.findall(r"a*b", teks))  # ['ab', 'aab', 'aaab', 'aaaab', 'aaaaab']

# + - 1 atau lebih
print("\n2. Quantifier + (1 atau lebih):")
print(re.findall(r"a+b", teks))  # ['ab', 'aab', 'aaab', 'aaaab', 'aaaaab']

# ? - 0 atau 1
print("\n3. Quantifier ? (0 atau 1):")
print(re.findall(r"a?b", "ab aab aaab"))  # ['ab', 'ab', 'ab']

# {n} - tepat n kali
print("\n4. Quantifier {n} (tepat n kali):")
print(re.findall(r"a{2}b", teks))  # ['aab', 'aaab', 'aaaab', 'aaaaab']

# {n,} - n atau lebih
print("\n5. Quantifier {n,} (n atau lebih):")
print(re.findall(r"a{2,}b", teks))  # ['aab', 'aaab', 'aaaab', 'aaaaab']

# {n,m} - antara n dan m
print("\n6. Quantifier {n,m} (antara n dan m):")
print(re.findall(r"a{2,3}b", teks))  # ['aab', 'aaab']

# Contoh parsing nomor telepon
telepon_teks = "Telepon: 08123456789, 021-1234567, +6281234567890"

# Pola untuk nomor Indonesia
pola_telepon = r"(\+?62|0)[0-9]{7,12}"
nomor_telepon = re.findall(pola_telepon, telepon_teks)
print(f"\nNomor telepon ditemukan: {nomor_telepon}")

# Contoh parsing tanggal
tanggal_teks = "Tanggal: 25/12/2023, 1/1/2024, 31/02/2023"

# Pola DD/MM/YYYY
pola_tanggal = r"\b\d{1,2}/\d{1,2}/\d{4}\b"
tanggal = re.findall(pola_tanggal, tanggal_teks)
print(f"Tanggal ditemukan: {tanggal}")

# Greedy vs Non-greedy
html = "
Hello
World
"
greedy = re.findall(r"
.*
", html)
non_greedy = re.findall(r"
.*?
", html)

print(f"\nGreedy: {greedy}")
print(f"Non-greedy: {non_greedy}")

5Ô∏è‚É£ Groups dan Capturing

Menggunakan parentheses untuk grouping.

import re

teks = "Nama: John Doe, Email: john@example.com, Telepon: 08123456789"

# Capturing groups dengan ()
hasil = re.search(r"Nama: (\w+) (\w+)", teks)
if hasil:
    print("1. Capturing groups:")
    print(f"Nama lengkap: {hasil.group()}")
    print(f"Nama depan: {hasil.group(1)}")
    print(f"Nama belakang: {hasil.group(2)}")

# Multiple groups
hasil_email = re.search(r"Email: (\w+)@(\w+)\.(\w+)", teks)
if hasil_email:
    print("\n2. Multiple groups:")
    print(f"Email lengkap: {hasil_email.group()}")
    print(f"Username: {hasil_email.group(1)}")
    print(f"Domain: {hasil_email.group(2)}")
    print(f"TLD: {hasil_email.group(3)}")

# Non-capturing groups dengan (?:)
hasil_telepon = re.search(r"Telepon: (?:\+\d{2})?(\d+)", teks)
if hasil_telepon:
    print("\n3. Non-capturing groups:")
    print(f"Nomor telepon: {hasil_telepon.group(1)}")

# Alternation dengan |
buah = "apel jeruk pisang"
pola_buah = r"apel|jeruk|pisang"
hasil_buah = re.findall(pola_buah, buah)
print(f"\n4. Alternation: {hasil_buah}")

# Backreferences - merujuk ke group sebelumnya
teks_duplikat = "kata kata, rumah rumah, buku buku"
pola_duplikat = r"(\w+) \1"  # \1 merujuk ke group 1
hasil_duplikat = re.findall(pola_duplikat, teks_duplikat)
print(f"5. Backreferences (kata duplikat): {hasil_duplikat}")

# Named groups dengan (?P)
hasil_named = re.search(r"Nama: (?P\w+) (?P\w+)", teks)
if hasil_named:
    print("\n6. Named groups:")
    print(f"Nama depan: {hasil_named.group('depan')}")
    print(f"Nama belakang: {hasil_named.group('belakang')}")

# Menggunakan named groups dalam findall
pola_named = r"(?P\w+)@(?P\w+)\.(?P\w+)"
hasil_named_all = re.findall(pola_named, "user1@domain.com user2@site.org")
print(f"7. Named groups in findall: {hasil_named_all}")

6Ô∏è‚É£ Flags dan Modifiers

Mengubah perilaku pencarian regex.

import re

teks = """Python adalah bahasa programming.
PYTHON juga powerful.
python bisa digunakan untuk web development."""

# re.IGNORECASE atau re.I - case insensitive
print("1. Case insensitive:")
hasil_i = re.findall(r"python", teks, re.IGNORECASE)
print(f"Ditemukan: {hasil_i}")

# re.MULTILINE atau re.M - multiline mode
teks_multiline = """Baris pertama: python
Baris kedua: PYTHON
Baris ketiga: Java"""

print("\n2. Multiline mode:")
# Tanpa MULTILINE
hasil_no_m = re.findall(r"^Baris", teks_multiline)
print(f"Tanpa MULTILINE: {hasil_no_m}")

# Dengan MULTILINE
hasil_m = re.findall(r"^Baris", teks_multiline, re.MULTILINE)
print(f"Dengan MULTILINE: {hasil_m}")

# re.DOTALL atau re.S - dot matches newline
teks_dotall = "Mulai\nisi\ntutup"
print("\n3. DOTALL mode:")
# Tanpa DOTALL
hasil_no_s = re.findall(r"Mulai.*tutup", teks_dotall)
print(f"Tanpa DOTALL: {hasil_no_s}")

# Dengan DOTALL
hasil_s = re.findall(r"Mulai.*tutup", teks_dotall, re.DOTALL)
print(f"Dengan DOTALL: {hasil_s}")

# Kombinasi flags
print("\n4. Kombinasi flags:")
pola = r"python"
hasil_combine = re.findall(pola, teks, re.IGNORECASE | re.MULTILINE)
print(f"Dengan IGNORECASE dan MULTILINE: {hasil_combine}")

# Menggunakan compile untuk performa lebih baik
print("\n5. Compiled regex:")
compiled_pattern = re.compile(r"python", re.IGNORECASE)
hasil_compiled = compiled_pattern.findall(teks)
print(f"Dengan compiled pattern: {hasil_compiled}")

# Menggunakan dalam loop
print("\n6. Menggunakan dalam loop:")
for match in compiled_pattern.finditer(teks):
    print(f"Ditemukan '{match.group()}' pada posisi {match.start()}-{match.end()}")

7Ô∏è‚É£ Substitusi dan Splitting

Mengganti dan memisahkan teks dengan regex.

import re

teks = "Email saya adalah user@example.com dan user2@test.org"

# sub() - substitusi
print("1. Substitusi:")
hasil_sub = re.sub(r"\w+@\w+\.\w+", "[EMAIL]", teks)
print(f"Original: {teks}")
print(f"Substitusi: {hasil_sub}")

# sub() dengan function
def censor_email(match):
    email = match.group()
    username, domain = email.split("@")
    return f"{username[0]}***@{domain}"

hasil_sub_func = re.sub(r"\w+@\w+\.\w+", censor_email, teks)
print(f"Substitusi dengan function: {hasil_sub_func}")

# subn() - substitusi dengan count
hasil_subn = re.subn(r"\w+@\w+\.\w+", "[EMAIL]", teks)
print(f"Substitusi dengan count: {hasil_subn}")

# split() - pemisahan
print("\n2. Splitting:")
teks_split = "satu, dua; tiga: empat"
hasil_split = re.split(r"[,;:]\s*", teks_split)
print(f"Original: {teks_split}")
print(f"Split: {hasil_split}")

# Contoh parsing CSV sederhana
csv_data = "Nama,Umur,Kota\nJohn,25,Jakarta\nAlice,30,Bandung"
baris = csv_data.split("\n")
for baris_data in baris:
    kolom = re.split(r",", baris_data)
    print(f"Baris: {kolom}")

# Mengganti format tanggal
tanggal_teks = "Tanggal lahir: 25/12/1990, Tanggal mulai: 01/06/2020"
tanggal_baru = re.sub(r"(\d{2})/(\d{2})/(\d{4})", r"\3-\2-\1", tanggal_teks)
print(f"\n3. Mengganti format tanggal:")
print(f"Original: {tanggal_teks}")
print(f"Format baru: {tanggal_baru}")

# Membersihkan whitespace berlebih
teks_bersih = "Ini   adalah    teks   dengan    spasi    berlebih"
teks_clean = re.sub(r"\s+", " ", teks_bersih)
print(f"\n4. Membersihkan whitespace:")
print(f"Original: '{teks_bersih}'")
print(f"Clean: '{teks_clean}'")

8Ô∏è‚É£ Contoh Penggunaan

Program validasi dan parsing data.

import re

class DataValidator:
    def __init__(self):
        # Compile patterns untuk performa
        self.email_pattern = re.compile(r"^[\w\.-]+@[\w\.-]+\.\w+$")
        self.phone_pattern = re.compile(r"^(\+62|62|0)[0-9]{7,12}$")
        self.date_pattern = re.compile(r"^\d{1,2}/\d{1,2}/\d{4}$")
        self.url_pattern = re.compile(r"^https?://[\w\.-]+\.\w+(?:/\S*)?$")
    
    def validate_email(self, email):
        return bool(self.email_pattern.match(email))
    
    def validate_phone(self, phone):
        return bool(self.phone_pattern.match(phone))
    
    def validate_date(self, date):
        return bool(self.date_pattern.match(date))
    
    def validate_url(self, url):
        return bool(self.url_pattern.match(url))
    
    def extract_emails(self, text):
        """Ekstrak semua email dari teks"""
        return self.email_pattern.findall(text)
    
    def extract_phones(self, text):
        """Ekstrak semua nomor telepon dari teks"""
        return self.phone_pattern.findall(text)
    
    def extract_dates(self, text):
        """Ekstrak semua tanggal dari teks"""
        return self.date_pattern.findall(text)

# Contoh penggunaan
validator = DataValidator()

# Test validasi
test_data = {
    "emails": ["user@example.com", "invalid-email", "user.name@domain.co.uk"],
    "phones": ["08123456789", "+6281234567890", "021-1234567", "invalid-phone"],
    "dates": ["25/12/2023", "1/1/2024", "31/02/2023", "invalid-date"],
    "urls": ["https://example.com", "http://test.org/path", "invalid-url"]
}

print("=== VALIDASI DATA ===")
for category, items in test_data.items():
    print(f"\n{category.upper()}:")
    for item in items:
        if category == "emails":
            valid = validator.validate_email(item)
        elif category == "phones":
            valid = validator.validate_phone(item)
        elif category == "dates":
            valid = validator.validate_date(item)
        elif category == "urls":
            valid = validator.validate_url(item)
        print(f"  {item}: {'‚úì' if valid else '‚úó'}")

# Test ekstraksi

# Contoh parsing log sederhana
log_data = """
2023-12-01 10:30:45 INFO User login: user@example.com from 192.168.1.1
2023-12-01 10:31:22 ERROR Invalid password for user@test.org
2023-12-01 10:32:15 INFO User login: admin@domain.com from 10.0.0.1
"""

def parse_log_entry(log_entry):
    """Parse single log entry"""
    pattern = r"(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) (\w+) (.+)"
    match = re.search(pattern, log_entry.strip())
    if match:
        timestamp, level, message = match.groups()
        return {
            "timestamp": timestamp,
            "level": level,
            "message": message
        }
    return None

print("\n=== PARSING LOG ===")
for line in log_data.strip().split("\n"):
    if line.strip():
        parsed = parse_log_entry(line)
        if parsed:
            print(f"[{parsed['timestamp']}] {parsed['level']}: {parsed['message']}")

# Contoh pembersihan teks
def clean_text(text):
    """Bersihkan teks dari karakter tidak diinginkan"""
    # Hapus multiple spaces
    text = re.sub(r"\s+", " ", text)
    # Hapus karakter non-alphanumeric kecuali spasi dan titik
    text = re.sub(r"[^\w\s\.]", "", text)
    # Trim spaces
    text = text.strip()
    return text
