2Ô∏è‚É£ Karakter Literal dan Meta

Karakter spesial regex: titik-titik ajaib! ‚ú®üîÆ

import re

text = "Halo Python123!"

# Literal: cocok persis
print(re.findall(r"Python", text))  # ['Python']

# . (dot): karakter apa saja
print(re.findall(r"P.th.n", text))  # ['Python']

# \d: angka 0-9
print(re.findall(r"\d+", text))  # ['123']

# \w: huruf/angka/underscore
print(re.findall(r"\w+", text))  # ['Halo', 'Python123']

3Ô∏è‚É£ Character Classes

Kumpulan karakter: pilih satu dari banyak! üéØ

import re

text = "Email: user@example.com, Phone: 08123456789"

# [abc] - salah satu huruf
print(re.findall(r"[aeiou]", text))  # ['a', 'e', 'a', 'e', 'o', 'e']

# [a-z] - huruf kecil
print(re.findall(r"[a-z]+", text))  # ['mail', 'user', 'example', 'com', 'hone']

# \d - angka (0-9)
print(re.findall(r"\d+", text))  # ['08123456789']

# \w - huruf/angka/underscore
print(re.findall(r"\w+", text))  # ['Email', 'user', 'example', 'com', 'Phone', '08123456789']

4Ô∏è‚É£ Quantifiers

Berapa banyak? Hitung kemunculan! üî¢

import re

text = "aa ab aab aaab"

# * - 0 atau lebih
print(re.findall(r"a*b", text))  # ['ab', 'aab', 'aaab']

# + - 1 atau lebih
print(re.findall(r"a+b", text))  # ['ab', 'aab', 'aaab']

# ? - 0 atau 1
print(re.findall(r"a?b", text))  # ['ab', 'ab', 'ab']

# {2} - tepat 2 kali
print(re.findall(r"a{2}b", text))  # ['aab']

# {2,3} - 2 sampai 3 kali
print(re.findall(r"a{2,3}b", text))  # ['aab', 'aaab']

5Ô∏è‚É£ Groups dan Capturing

Kelompok tangkap: ambil bagian spesifik! üé£

import re

text = "Email: user@example.com"

# () - capturing group
match = re.search(r"(\w+)@(\w+)\.(\w+)", text)
if match:
    print(match.group(1))  # user
    print(match.group(2))  # example
    print(match.group(3))  # com

# (?:) - non-capturing group
match2 = re.search(r"(?:https?://)?(\w+\.\w+)", "https://example.com")
if match2:
    print(match2.group(1))  # example.com

6Ô∏è‚É£ Flags dan Modifiers

Mode khusus: ubah cara kerja regex! ‚öôÔ∏è

import re

text = "Python PYTHON python"

# re.I - ignore case
print(re.findall(r"python", text, re.I))  # ['Python', 'PYTHON', 'python']

# re.M - multiline
text2 = "line1\nline2\nline3"
print(re.findall(r"^line", text2, re.M))  # ['line1', 'line2', 'line3']

# re.S - dot matches newline
text3 = "start\nend"
print(re.findall(r"start.*end", text3, re.S))  # ['start\nend']

7Ô∏è‚É£ Substitusi dan Splitting

Ganti dan bagi teks: sihir transformasi! ‚ú®üîÄ

import re

text = "Contact: user@example.com"

# sub() - replace
print(re.sub(r"\w+@\w+\.\w+", "[EMAIL]", text))  # Contact: [EMAIL]

# split() - split text
text2 = "apple,banana;orange"
print(re.split(r"[,;]", text2))  # ['apple', 'banana', 'orange']

# Replace with function
def censor(match):
    return match.group()[0] + "***"

print(re.sub(r"\w+", censor, "secret word"))  # s*** w***

8Ô∏è‚É£ Contoh Penggunaan

Regex dalam aksi: validasi cepat dan ekstraksi data! üöÄ

import re

# Validasi email sederhana
def is_valid_email(email):
    pattern = r"^[\w\.-]+@[\w\.-]+\.\w+$"
    return bool(re.match(pattern, email))

print(is_valid_email("user@example.com"))  # True

# Ekstrak nomor telepon
text = "Call me at 08123456789 or +628123456789"
phones = re.findall(r"(\+62|0)\d{7,12}", text)
print(phones)  # ['08123456789', '+628123456789']

# Bersihkan teks
dirty = "  Hello!!!   World???  "
clean = re.sub(r"[^\w\s]", "", dirty).strip()
print(clean)  # Hello World

9Ô∏è‚É£ Latihan Praktis

Coba sendiri: validasi email dan ekstrak nomor! üí™

import re

# Tantangan: Buat fungsi validasi email
def validate_email(email):
    pattern = r"^[\w\.-]+@[\w\.-]+\.\w+$"
    return bool(re.match(pattern, email))

# Test
print(validate_email("user@example.com"))  # True
print(validate_email("invalid-email"))     # False

# Tantangan: Ekstrak semua angka dari teks
text = "Umur saya 25 tahun, tinggi 170 cm"
numbers = re.findall(r"\d+", text)
print(numbers)  # ['25', '170']
