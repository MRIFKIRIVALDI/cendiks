Pelajari cara fungsi mengembalikan nilai

Materi Pembelajaran

1Ô∏è‚É£ Return Statement

Mengembalikan nilai dari fungsi.

def tambah(a, b):
    hasil = a + b
    return hasil

# Menggunakan nilai return
result = tambah(5, 3)
print(result)  # Output: 8
print(tambah(10, 7))  # Output: 17

2Ô∏è‚É£ Mengembalikan Nilai

return nilai.

def kuadrat(x):
    return x * x

def salam(nama):
    return f"Halo, {nama}!"

# Menggunakan return value
angka = 5
hasil_kuadrat = kuadrat(angka)
print(f"Kuadrat dari {angka} adalah {hasil_kuadrat}")

pesan = salam("John")
print(pesan)

3Ô∏è‚É£ Multiple Return Values

return val1, val2.

def hitung_luas_keliling_persegi(sisi):
    luas = sisi * sisi
    keliling = 4 * sisi
    return luas, keliling

# Menggunakan multiple return values
luas, keliling = hitung_luas_keliling_persegi(5)
print(f"Luas: {luas}")       # Output: Luas: 25
print(f"Keliling: {keliling}")  # Output: Keliling: 20

# Atau menyimpan dalam tuple
hasil = hitung_luas_keliling_persegi(5)
print(f"Hasil: {hasil}")     # Output: Hasil: (25, 20)
print(f"Luas: {hasil[0]}")   # Output: Luas: 25

4Ô∏è‚É£ Return Tanpa Nilai

Hanya return, atau None.

def cetak_info(nama):
    print(f"Nama: {nama}")
    return  # Keluar dari fungsi

def fungsi_tanpa_return():
    print("Ini fungsi tanpa return")

# Menggunakan fungsi dengan return saja
cetak_info("John")
print("Baris ini akan dieksekusi")

# Fungsi tanpa return mengembalikan None
result = fungsi_tanpa_return()
print(f"Return value: {result}")  # Output: Return value: None

5Ô∏è‚É£ Contoh Penggunaan

Fungsi kalkulator yang mengembalikan hasil.

def kalkulator(a, b, operasi):
    if operasi == "+":
        return a + b
    elif operasi == "-":
        return a - b
    elif operasi == "*":
        return a * b
    elif operasi == "/":
        if b != 0:
            return a / b
        else:
            return "Error: Division by zero"
    else:
        return "Error: Operasi tidak valid"

# Menggunakan kalkulator
print(kalkulator(10, 5, "+"))   # Output: 15
print(kalkulator(10, 5, "-"))   # Output: 5
print(kalkulator(10, 5, "*"))   # Output: 50
print(kalkulator(10, 5, "/"))   # Output: 2.0
print(kalkulator(10, 0, "/"))   # Output: Error: Division by zero
