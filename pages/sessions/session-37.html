7ï¸âƒ£ Methods dalam re Module

Regex methods: cari, ganti, split! ğŸ”âœ¨

import re

text = "Hello Python World!"

# Cari dari awal
match = re.match(r'Hello', text)
print(match.group())  # Hello

# Cari di mana saja
search = re.search(r'Python', text)
print(search.group())  # Python

# Cari semua
words = re.findall(r'\w+', text)
print(words)  # ['Hello', 'Python', 'World']

# Ganti
new_text = re.sub(r'World', 'Universe', text)
print(new_text)  # Hello Python Universe!
