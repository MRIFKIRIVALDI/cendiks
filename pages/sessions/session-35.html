2Ô∏è‚É£ Mode File

Berbagai mode untuk membuka file.

# Mode-mode file:
# "r"  - Read (default) - baca file
# "w"  - Write - tulis file (overwrite jika ada)
# "a"  - Append - tambah ke akhir file
# "x"  - Create - buat file baru (error jika sudah ada)
# "r+" - Read + Write
# "w+" - Write + Read (overwrite)
# "a+" - Append + Read

# Text mode (default)
with open("file.txt", "r") as f:
    pass

# Binary mode
with open("file.jpg", "rb") as f:
    pass

# Specify encoding
with open("file.txt", "r", encoding="utf-8") as f:
    pass

3Ô∏è‚É£ Membaca File

Berbagai cara membaca isi file.

# Membuat file contoh terlebih dahulu
with open("contoh.txt", "w") as f:
    f.write("Baris pertama\n")
    f.write("Baris kedua\n")
    f.write("Baris ketiga\n")

# 1. Membaca seluruh file
with open("contoh.txt", "r") as file:
    isi = file.read()
    print("Seluruh isi file:")
    print(repr(isi))

# 2. Membaca per baris
with open("contoh.txt", "r") as file:
    print("\nMembaca per baris:")
    for baris in file:
        print(repr(baris))

# 3. Membaca semua baris ke list
with open("contoh.txt", "r") as file:
    baris_list = file.readlines()
    print("\nList baris:")
    print(baris_list)

# 4. Membaca baris tertentu
with open("contoh.txt", "r") as file:
    baris_pertama = file.readline()
    baris_kedua = file.readline()
    print(f"\nBaris pertama: {repr(baris_pertama)}")
    print(f"Baris kedua: {repr(baris_kedua)}")

5Ô∏è‚É£ File Position

tell() dan seek() untuk navigasi posisi file.

with open("contoh.txt", "r") as file:
    # Posisi awal
    print(f"Posisi awal: {file.tell()}")

    # Membaca beberapa karakter
    data = file.read(10)
    print(f"Data: {repr(data)}")
    print(f"Posisi setelah baca: {file.tell()}")

    # Pindah ke posisi tertentu
    file.seek(0)  # Kembali ke awal
    print(f"Posisi setelah seek(0): {file.tell()}")

    # Seek dari akhir file
    file.seek(-10, 2)  # 10 karakter dari akhir
    data_akhir = file.read()
    print(f"10 karakter terakhir: {repr(data_akhir)}")

    # Seek relatif dari posisi saat ini
    file.seek(5, 1)  # 5 karakter maju dari posisi saat ini
    data_relatif = file.read(5)
    print(f"Data relatif: {repr(data_relatif)}")

6Ô∏è‚É£ Error Handling

Menangani error saat bekerja dengan file.

def baca_file(nama_file):
    try:
        with open(nama_file, "r") as file:
            return file.read()
    except FileNotFoundError:
        print(f"Error: File '{nama_file}' tidak ditemukan")
        return None
    except PermissionError:
        print(f"Error: Tidak ada izin untuk membaca '{nama_file}'")
        return None
    except Exception as e:
        print(f"Error tidak terduga: {e}")
        return None

# Menggunakan fungsi
isi = baca_file("contoh.txt")
if isi:
    print("Isi file:")

import os

nama_file = "data/contoh.txt"

# Mendapatkan direktori
direktori = os.path.dirname(nama_file)
print(f"Direktori: {direktori}")

print(f"Nama file: {nama}")

# Gabung path
path_lengkap = os.path.join("folder", "subfolder", "file.txt")
print(f"Path lengkap: {path_lengkap}")

# Normalisasi path
path = "folder/../data/./file.txt"
normalized = os.path.normpath(path)
print(f"Path normalized: {normalized}")

# Menggunakan pathlib (Python 3.4+)
path_obj = Path("data/contoh.txt")
print(f"Parent: {path_obj.parent}")
print(f"Name: {path_obj.name}")
print(f"Suffix: {path_obj.suffix}")

# Operasi path dengan pathlib
new_path = path_obj.with_suffix(".bak")
print(f"New path: {new_path}")

# Cek apakah file ada
if path_obj.exists():
    print("File ada")
else:
    print("File tidak ada")

8Ô∏è‚É£ Contoh Penggunaan

Program sederhana untuk mengelola daftar tugas.

def load_tasks(filename="tasks.txt"):
    """Memuat daftar tugas dari file"""
    try:
        with open(filename, "r") as file:
            tasks = [line.strip() for line in file if line.strip()]
        return tasks
    except FileNotFoundError:
        return []

def save_tasks(tasks, filename="tasks.txt"):
    """Menyimpan daftar tugas ke file"""
    with open(filename, "w") as file:
        for task in tasks:
            file.write(task + "\n")

def add_task(tasks, task):
    """Menambah tugas baru"""
    tasks.append(task)
    print(f"Tugas '{task}' ditambahkan")

def remove_task(tasks, index):
    """Menghapus tugas berdasarkan index"""
    if 0 <= index < len(tasks):
        removed = tasks.pop(index)
        print(f"Tugas '{removed}' dihapus")
        return True
    else:
        print("Index tidak valid")
        return False

def show_tasks(tasks):
    """Menampilkan semua tugas"""
    if not tasks:
        print("Tidak ada tugas")
        return

    print("Daftar Tugas:")
    for i, task in enumerate(tasks, 1):
        print(f"{i}. {task}")

def main():
    tasks = load_tasks()

    while True:
        print("\n=== APLIKASI DAFTAR TUGAS ===")
        print("1. Tampilkan tugas")
        print("2. Tambah tugas")
        print("3. Hapus tugas")
        print("4. Simpan dan keluar")

        try:
            pilihan = input("Pilih menu (1-4): ")

            if pilihan == "1":
                show_tasks(tasks)
            elif pilihan == "2":
                task = input("Masukkan tugas baru: ")
                if task.strip():
                    add_task(tasks, task.strip())
                else:
                    print("Tugas tidak boleh kosong")
            elif pilihan == "3":
                show_tasks(tasks)
                if tasks:
                    index = int(input("Masukkan nomor tugas yang akan dihapus: ")) - 1
                    remove_task(tasks, index)
            elif pilihan == "4":
                save_tasks(tasks)
                print("Tugas disimpan. Terima kasih!")
                break
            else:
                print("Pilihan tidak valid")

        except ValueError:
            print("Masukkan angka yang valid")
        except KeyboardInterrupt:
            save_tasks(tasks)
            print("\nProgram dihentikan. Tugas disimpan.")
            break

if __name__ == "__main__":
    main()
