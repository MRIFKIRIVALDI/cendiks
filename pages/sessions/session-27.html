üìö Sesi 27: Cara Menggunakan Parameter & Argument pada Fungsi Python

Pelajari berbagai cara menggunakan parameter dan argument

Materi Pembelajaran

1Ô∏è‚É£ Positional Arguments

Argument berdasarkan posisi parameter.

def perkenalan(nama, umur):
    print(f"Halo, nama saya {nama} dan saya berumur {umur} tahun.")

# Memanggil dengan positional arguments
perkenalan("John", 25)    # Output: Halo, nama saya John dan saya berumur 25 tahun.
perkenalan("Alice", 30)   # Output: Halo, nama saya Alice dan saya berumur 30 tahun.

# Urutan penting!
perkenalan(25, "John")    # Output: Halo, nama saya 25 dan saya berumur John tahun. (salah!)

2Ô∏è‚É£ Keyword Arguments

Argument dengan nama parameter.

def perkenalan(nama, umur):
    print(f"Halo, nama saya {nama} dan saya berumur {umur} tahun.")

# Memanggil dengan keyword arguments
perkenalan(nama="John", umur=25)    # Output: Halo, nama saya John dan saya berumur 25 tahun.
perkenalan(umur=30, nama="Alice")   # Urutan tidak penting!

# Campuran positional dan keyword
perkenalan("Bob", umur=35)          # Output: Halo, nama saya Bob dan saya berumur 35 tahun.

3Ô∏è‚É£ Default Parameters

Parameter dengan nilai default.

def sapa(nama, salam="Halo"):
    print(f"{salam}, {nama}!")

# Memanggil tanpa argument untuk parameter default
sapa("John")                    # Output: Halo, John!
sapa("Alice", "Selamat pagi")   # Output: Selamat pagi, Alice!

# Default parameter harus di akhir
def hitung_luas(panjang, lebar=1):  # lebar default 1
    return panjang * lebar

print(hitung_luas(5))      # Output: 5 (5 * 1)
print(hitung_luas(5, 3))   # Output: 15 (5 * 3)

4Ô∏è‚É£ *args (Arbitrary Positional Arguments)

Parameter untuk jumlah argument yang tidak terbatas.

def jumlahkan(*args):
    total = 0
    for angka in args:
        total += angka
    return total

# Memanggil dengan berbagai jumlah argument
print(jumlahkan(1, 2, 3))        # Output: 6
print(jumlahkan(10, 20))         # Output: 30
print(jumlahkan(1, 2, 3, 4, 5))  # Output: 15

# args adalah tuple
def cetak_args(*args):
    print(f"Jumlah args: {len(args)}")
    print(f"Args: {args}")
    print(f"Tipe: {type(args)}")

cetak_args(1, "hello", 3.14, True)

5Ô∏è‚É£ **kwargs (Arbitrary Keyword Arguments)

Parameter untuk keyword arguments yang tidak terbatas.

def info_mahasiswa(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

# Memanggil dengan berbagai keyword arguments
info_mahasiswa(nama="John", umur=20, jurusan="Informatika")
# Output:
# nama: John
# umur: 20
# jurusan: Informatika

# kwargs adalah dictionary
def cetak_kwargs(**kwargs):
    print(f"Jumlah kwargs: {len(kwargs)}")
    print(f"Kwargs: {kwargs}")
    print(f"Tipe: {type(kwargs)}")

cetak_kwargs(nama="Alice", kota="Jakarta", hobi="Coding")

6Ô∏è‚É£ Kombinasi Parameter

Menggabungkan berbagai jenis parameter.

def fungsi_lengkap(nama, umur=18, *hobi, **info_tambahan):
    print(f"Nama: {nama}")
    print(f"Umur: {umur}")
    print(f"Hobi: {hobi}")
    print(f"Info tambahan: {info_tambahan}")

# Memanggil dengan kombinasi parameter
fungsi_lengkap("John", 25, "coding", "gaming", kota="Jakarta", pekerjaan="Programmer")
# Output:
# Nama: John
# Umur: 25
# Hobi: ('coding', 'gaming')
# Info tambahan: {'kota': 'Jakarta', 'pekerjaan': 'Programmer'}

7Ô∏è‚É£ Contoh Penggunaan

Fungsi untuk membuat profil mahasiswa.

def buat_profil(nama, umur, jurusan, *nilai, **info_tambahan):
    """Fungsi untuk membuat profil mahasiswa"""
    profil = {
        "nama": nama,
        "umur": umur,
        "jurusan": jurusan,
        "nilai": nilai,
        "rata_rata": sum(nilai) / len(nilai) if nilai else 0
    }

    # Menambahkan info tambahan
    profil.update(info_tambahan)

    return profil

# Menggunakan fungsi
profil1 = buat_profil("Alice", 20, "Informatika", 85, 90, 88, kota="Jakarta", hobi="Coding")
profil2 = buat_profil("Bob", 21, "Sistem Informasi", 78, 82, 85, 80, kota="Bandung")

print("Profil Alice:")
for key, value in profil1.items():
    print(f"  {key}: {value}")

print("\nProfil Bob:")
for key, value in profil2.items():
    print(f"  {key}: {value}")
