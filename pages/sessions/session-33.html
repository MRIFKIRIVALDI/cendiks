5ï¸âƒ£ Fungsi Kombinatorik

Permutasi dan kombinasi - hitung peluang dan kemungkinan! ğŸ²

import math

# Faktorial
print(f"5! = {math.factorial(5)}")  # Output: 120

# Permutasi P(n,r) = n! / (n-r)!
def permutasi(n, r):
    return math.factorial(n) // math.factorial(n - r)

# Kombinasi C(n,r) = n! / (r! * (n-r)!)
def kombinasi(n, r):
    return math.factorial(n) // (math.factorial(r) * math.factorial(n - r))

# Contoh penggunaan
n, r = 5, 3
print(f"P({n},{r}) = {permutasi(n, r)}")  # Output: 60
print(f"C({n},{r}) = {kombinasi(n, r)}")  # Output: 10

# Lotere: 6 angka dari 49
total_kombinasi = kombinasi(49, 6)
print(f"Total kombinasi lotere: {total_kombinasi:,}")

9ï¸âƒ£ Latihan Praktis

Hitung jarak antara dua titik dengan math.hypot - seperti GPS! ğŸ“

import math

def jarak_antar_titik(x1, y1, x2, y2):
    return math.hypot(x2 - x1, y2 - y1)

# Hitung jarak dari rumah ke sekolah
jarak = jarak_antar_titik(0, 0, 3, 4)
print(f"Jarak rumah ke sekolah: {jarak:.1f} km")

# Hitung jarak tempuh sepeda
kecepatan = 15  # km/jam
waktu = jarak / kecepatan
print(f"Waktu tempuh: {waktu:.1f} jam")
