5Ô∏è‚É£ Fungsi Kombinatorik

Permutasi dan kombinasi.

import math

# Faktorial
print(f"5! = {math.factorial(5)}")  # Output: 120

# Permutasi P(n,r) = n! / (n-r)!
def permutasi(n, r):
    return math.factorial(n) // math.factorial(n - r)

# Kombinasi C(n,r) = n! / (r! * (n-r)!)
def kombinasi(n, r):
    return math.factorial(n) // (math.factorial(r) * math.factorial(n - r))

# Contoh penggunaan
n, r = 5, 3
print(f"P({n},{r}) = {permutasi(n, r)}")  # Output: 60
print(f"C({n},{r}) = {kombinasi(n, r)}")  # Output: 10

# Lotere: 6 angka dari 49
total_kombinasi = kombinasi(49, 6)
print(f"Total kombinasi lotere: {total_kombinasi:,}")

8Ô∏è‚É£ Contoh Penggunaan

Program kalkulator ilmiah sederhana.

import math

def kalkulator_ilmiah():
    """Kalkulator ilmiah sederhana"""
    print("=== KALKULATOR ILMIAH ===")
    print("Pilih operasi:")
    print("1. Akar kuadrat")
    print("2. Pangkat")
    print("3. Logaritma natural")
    print("4. Logaritma basis 10")
    print("5. Sinus")
    print("6. Cosinus")
    print("7. Tangen")
    print("8. Faktorial")
    print("9. Keluar")

    while True:
        try:
            pilihan = input("\nPilih operasi (1-9): ")

            if pilihan == "9":
                print("Terima kasih!")
                break

            if pilihan in ["1", "2", "3", "4", "8"]:
                angka = float(input("Masukkan angka: "))
            elif pilihan in ["5", "6", "7"]:
                derajat = float(input("Masukkan sudut (derajat): "))
                radian = math.radians(derajat)

            if pilihan == "1":
                if angka >= 0:
                    hasil = math.sqrt(angka)
                    print(f"‚àö{angka} = {hasil:.4f}")
                else:
                    print("Error: Angka negatif tidak bisa diakar kuadratkan")
            elif pilihan == "2":
                pangkat = int(input("Masukkan pangkat: "))
                hasil = math.pow(angka, pangkat)
                print(f"{angka}^{pangkat} = {hasil}")
            elif pilihan == "3":
                if angka > 0:
                    hasil = math.log(angka)
                    print(f"ln({angka}) = {hasil:.4f}")
                else:
                    print("Error: Logaritma hanya untuk angka positif")
            elif pilihan == "4":
                if angka > 0:
                    hasil = math.log10(angka)
                    print(f"log10({angka}) = {hasil:.4f}")
                else:
                    print("Error: Logaritma hanya untuk angka positif")
            elif pilihan == "5":
                hasil = math.sin(radian)
                print(f"sin({derajat}¬∞) = {hasil:.4f}")
            elif pilihan == "6":
                hasil = math.cos(radian)
                print(f"cos({derajat}¬∞) = {hasil:.4f}")
            elif pilihan == "7":
                hasil = math.tan(radian)
                print(f"tan({derajat}¬∞) = {hasil:.4f}")
            elif pilihan == "8":
                if angka >= 0 and angka == int(angka):
                    hasil = math.factorial(int(angka))
                    print(f"{int(angka)}! = {hasil}")
                else:
                    print("Error: Faktorial hanya untuk bilangan bulat non-negatif")

        except ValueError:
            print("Error: Masukkan angka yang valid!")
        except Exception as e:
            print(f"Error: {e}")

def main():
    kalkulator_ilmiah()

if __name__ == "__main__":
    main()
