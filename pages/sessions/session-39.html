import tkinter as tk
from tkinter import messagebox

root = tk.Tk()
root.title("Event Handling")
root.geometry("500x400")

# Label untuk menampilkan informasi
info_label = tk.Label(root, text="Informasi event akan muncul di sini", font=("Arial", 10))
info_label.pack(pady=10)

# === MOUSE EVENTS ===
def on_click(event):
    info_label.config(text=f"Mouse diklik pada posisi ({event.x}, {event.y})")

def on_double_click(event):
    messagebox.showinfo("Double Click", "Double click terdeteksi!")

def on_enter(event):
    info_label.config(text="Mouse masuk ke area button")

def on_leave(event):
    info_label.config(text="Mouse keluar dari area button")

button = tk.Button(root, text="Klik atau Hover Saya")
button.pack(pady=10)

# Bind mouse events
button.bind("", on_click)  # Left click
button.bind("", on_double_click)  # Double click
button.bind("", on_enter)  # Mouse enter
button.bind("", on_leave)  # Mouse leave

# === KEYBOARD EVENTS ===
def on_key_press(event):
    key = event.keysym
    info_label.config(text=f"Key pressed: {key}")

def on_key_release(event):
    key = event.keysym
    info_label.config(text=f"Key released: {key}")

# Entry untuk keyboard events
entry = tk.Entry(root, width=30)
entry.pack(pady=10)
entry.insert(0, "Ketik sesuatu...")

entry.bind("", on_key_press)
entry.bind("", on_key_release)

# === WINDOW EVENTS ===
def on_resize(event):
    width = event.width
    height = event.height
    info_label.config(text=f"Window resized: {width}x{height}")

root.bind("", on_resize)

# === TIMER EVENTS ===
counter = 0
def update_counter():
    global counter
    counter += 1
    timer_label.config(text=f"Timer: {counter} detik")
    root.after(1000, update_counter)  # Call again after 1 second

timer_label = tk.Label(root, text="Timer: 0 detik", font=("Arial", 12))
timer_label.pack(pady=10)

# Start timer
update_counter()

# === FOCUS EVENTS ===
def on_focus_in(event):
    info_label.config(text="Entry mendapat focus")

def on_focus_out(event):
    info_label.config(text="Entry kehilangan focus")

entry.bind("", on_focus_in)
entry.bind("", on_focus_out)

root.mainloop()

Contoh aplikasi GUI lengkap dengan Tkinter.

import tkinter as tk
from tkinter import ttk, messagebox, filedialog
import json
import os

class ContactManager:
    def __init__(self, root):
        self.root = root
        self.root.title("Contact Manager - Tkinter App")
        self.root.geometry("600x500")

        # Data storage
        self.contacts = []
        self.load_contacts()

        # Create GUI
        self.create_widgets()
        self.update_contact_list()

    def create_widgets(self):
        # Main frame
        main_frame = tk.Frame(self.root)
        main_frame.pack(fill=tk.BOTH, expand=True, padx=10, pady=10)

        # Title
        title_label = tk.Label(main_frame, text="Contact Manager",
                              font=("Arial", 16, "bold"))
        title_label.pack(pady=10)

        # Input frame
        input_frame = tk.Frame(main_frame)
        input_frame.pack(fill=tk.X, pady=10)

        # Name
        tk.Label(input_frame, text="Name:").grid(row=0, column=0, sticky="e", padx=5, pady=2)
        self.name_entry = tk.Entry(input_frame, width=30)
        self.name_entry.grid(row=0, column=1, padx=5, pady=2)

        # Phone
        tk.Label(input_frame, text="Phone:").grid(row=1, column=0, sticky="e", padx=5, pady=2)
        self.phone_entry = tk.Entry(input_frame, width=30)
        self.phone_entry.grid(row=1, column=1, padx=5, pady=2)

        # Email
        tk.Label(input_frame, text="Email:").grid(row=2, column=0, sticky="e", padx=5, pady=2)
        self.email_entry = tk.Entry(input_frame, width=30)
        self.email_entry.grid(row=2, column=1, padx=5, pady=2)

        # Buttons
        button_frame = tk.Frame(input_frame)
        button_frame.grid(row=3, column=0, columnspan=2, pady=10)

        tk.Button(button_frame, text="Add Contact", command=self.add_contact).pack(side=tk.LEFT, padx=5)
        tk.Button(button_frame, text="Update Contact", command=self.update_contact).pack(side=tk.LEFT, padx=5)
        tk.Button(button_frame, text="Delete Contact", command=self.delete_contact).pack(side=tk.LEFT, padx=5)
        tk.Button(button_frame, text="Clear", command=self.clear_form).pack(side=tk.LEFT, padx=5)

        # Search frame
        search_frame = tk.Frame(main_frame)
        search_frame.pack(fill=tk.X, pady=5)

        tk.Label(search_frame, text="Search:").pack(side=tk.LEFT, padx=5)
        self.search_entry = tk.Entry(search_frame, width=30)
        self.search_entry.pack(side=tk.LEFT, padx=5)
        self.search_entry.bind("", self.search_contacts)

        # Contact list
        list_frame = tk.Frame(main_frame)
        list_frame.pack(fill=tk.BOTH, expand=True, pady=10)

        # Scrollbar
        scrollbar = tk.Scrollbar(list_frame)
        scrollbar.pack(side=tk.RIGHT, fill=tk.Y)

        # Listbox
        self.contact_listbox = tk.Listbox(list_frame, height=10,
                                         yscrollcommand=scrollbar.set)
        self.contact_listbox.pack(fill=tk.BOTH, expand=True)
        scrollbar.config(command=self.contact_listbox.yview)

        # Bind selection event
        self.contact_listbox.bind("<>", self.on_select_contact)

        # Status bar
        self.status_label = tk.Label(main_frame, text="Ready", relief=tk.SUNKEN, anchor="w")
        self.status_label.pack(fill=tk.X, side=tk.BOTTOM)

    def add_contact(self):
        name = self.name_entry.get().strip()
        phone = self.phone_entry.get().strip()
        email = self.email_entry.get().strip()

        if not name:
            messagebox.showerror("Error", "Name is required!")
            return

        contact = {
            "name": name,
            "phone": phone,
            "email": email
        }

        self.contacts.append(contact)
        self.save_contacts()
        self.update_contact_list()
        self.clear_form()
        self.status_label.config(text=f"Added contact: {name}")

    def update_contact(self):
        selection = self.contact_listbox.curselection()
        if not selection:
            messagebox.showerror("Error", "Please select a contact to update!")
            return

        index = selection[0]
        name = self.name_entry.get().strip()
        phone = self.phone_entry.get().strip()
        email = self.email_entry.get().strip()

        if not name:
            messagebox.showerror("Error", "Name is required!")
            return

        self.contacts[index] = {
            "name": name,
            "phone": phone,
            "email": email
        }

        self.save_contacts()
        self.update_contact_list()
        self.status_label.config(text=f"Updated contact: {name}")

    def delete_contact(self):
        selection = self.contact_listbox.curselection()
        if not selection:
            messagebox.showerror("Error", "Please select a contact to delete!")
            return

        index = selection[0]
        contact = self.contacts[index]

        if messagebox.askyesno("Confirm Delete", f"Delete contact '{contact['name']}'?"):
            del self.contacts[index]
            self.save_contacts()
            self.update_contact_list()
            self.clear_form()
            self.status_label.config(text=f"Deleted contact: {contact['name']}")

    def on_select_contact(self, event):
        selection = self.contact_listbox.curselection()
        if selection:
            index = selection[0]
            contact = self.contacts[index]

            self.name_entry.delete(0, tk.END)
            self.name_entry.insert(0, contact["name"])

            self.phone_entry.delete(0, tk.END)
            self.phone_entry.insert(0, contact["phone"])

            self.email_entry.delete(0, tk.END)
            self.email_entry.insert(0, contact["email"])

    def clear_form(self):
        self.name_entry.delete(0, tk.END)
        self.phone_entry.delete(0, tk.END)
        self.email_entry.delete(0, tk.END)
        self.contact_listbox.selection_clear(0, tk.END)

    def search_contacts(self, event):
        query = self.search_entry.get().lower()
        self.contact_listbox.delete(0, tk.END)

        for contact in self.contacts:
            if query in contact["name"].lower() or query in contact["phone"] or query in contact["email"].lower():
                display_text = f"{contact['name']} - {contact['phone']}"
                self.contact_listbox.insert(tk.END, display_text)

    def update_contact_list(self):
        self.contact_listbox.delete(0, tk.END)
        for contact in self.contacts:
            display_text = f"{contact['name']} - {contact['phone']}"
            self.contact_listbox.insert(tk.END, display_text)

    def load_contacts(self):
        try:
            if os.path.exists("contacts.json"):
                with open("contacts.json", "r") as f:
                    self.contacts = json.load(f)
        except Exception as e:
            messagebox.showerror("Error", f"Failed to load contacts: {e}")

    def save_contacts(self):
        try:
            with open("contacts.json", "w") as f:
                json.dump(self.contacts, f, indent=2)
        except Exception as e:
            messagebox.showerror("Error", f"Failed to save contacts: {e}")

def main():
    root = tk.Tk()
    app = ContactManager(root)
    root.mainloop()

if __name__ == "__main__":
    main()

import tkinter as tk
from tkinter import ttk
import threading
import time

# === BEST PRACTICES ===

class BestPracticesDemo:
    def __init__(self, root):
        self.root = root
        self.root.title("Tkinter Best Practices")
        self.root.geometry("500x600")

        self.create_widgets()

    def create_widgets(self):
        # 1. Use descriptive variable names
        main_frame = tk.Frame(self.root, padx=10, pady=10)
        main_frame.pack(fill=tk.BOTH, expand=True)

        # 2. Group related widgets in frames
        practice_frame = tk.LabelFrame(main_frame, text="Best Practices", padx=10, pady=10)
        practice_frame.pack(fill=tk.X, pady=5)

        # 3. Use consistent padding and spacing
        tk.Label(practice_frame, text="1. Use descriptive names and comments",
                font=("Arial", 10)).pack(anchor="w", pady=2)

        tk.Label(practice_frame, text="2. Group widgets logically in frames",
                font=("Arial", 10)).pack(anchor="w", pady=2)

        tk.Label(practice_frame, text="3. Use consistent spacing and padding",
                font=("Arial", 10)).pack(anchor="w", pady=2)

        tk.Label(practice_frame, text="4. Handle exceptions properly",
                font=("Arial", 10)).pack(anchor="w", pady=2)

        tk.Label(practice_frame, text="5. Use StringVar, IntVar, etc. for dynamic updates",
                font=("Arial", 10)).pack(anchor="w", pady=2)

        # 5. Example of using variables
        var_frame = tk.LabelFrame(main_frame, text="Using Variables", padx=10, pady=10)
        var_frame.pack(fill=tk.X, pady=5)

        # StringVar example
        self.text_var = tk.StringVar(value="Hello Tkinter!")
        tk.Label(var_frame, textvariable=self.text_var).pack(pady=5)

        tk.Button(var_frame, text="Change Text",
                 command=lambda: self.text_var.set("Text changed!")).pack(pady=5)

        # IntVar example
        self.counter_var = tk.IntVar(value=0)
        tk.Label(var_frame, textvariable=self.counter_var).pack(pady=5)

        tk.Button(var_frame, text="Increment",
                 command=self.increment_counter).pack(pady=5)

        # 6. Threading for long operations
        thread_frame = tk.LabelFrame(main_frame, text="Threading Example", padx=10, pady=10)
        thread_frame.pack(fill=tk.X, pady=5)

        self.progress_var = tk.StringVar(value="Ready")
        tk.Label(thread_frame, textvariable=self.progress_var).pack(pady=5)

        tk.Button(thread_frame, text="Start Long Operation",
                 command=self.start_long_operation).pack(pady=5)

        # 7. Proper error handling
        error_frame = tk.LabelFrame(main_frame, text="Error Handling", padx=10, pady=10)
        error_frame.pack(fill=tk.X, pady=5)

        tk.Button(error_frame, text="Test Error Handling",
                 command=self.test_error_handling).pack(pady=5)

        self.error_result = tk.StringVar(value="")
        tk.Label(error_frame, textvariable=self.error_result).pack(pady=5)

        # 8. Clean up resources
        self.root.protocol("WM_DELETE_WINDOW", self.on_closing)

    def increment_counter(self):
        current = self.counter_var.get()
        self.counter_var.set(current + 1)

    def start_long_operation(self):
        # Run long operation in separate thread
        thread = threading.Thread(target=self.long_operation)
        thread.daemon = True
        thread.start()

    def long_operation(self):
        # Simulate long operation
        for i in range(5):
            time.sleep(1)
            # Use after() to update GUI from thread
            self.root.after(0, lambda: self.progress_var.set(f"Processing... {i+1}/5"))

        self.root.after(0, lambda: self.progress_var.set("Completed!"))

    def test_error_handling(self):
        try:

# === COMMON MISTAKES TO AVOID ===

def demonstrate_mistakes():
    """Examples of common Tkinter mistakes"""

    root = tk.Tk()
    root.title("Common Mistakes")
    root.geometry("400x300")

    # Mistake 1: Not using mainloop() properly
    # Wrong: root.mainloop() inside a function that returns
    # Correct: mainloop() should be the last statement

    # Mistake 2: Modifying GUI from other threads directly
    # Wrong: Update widgets directly from thread
    # Correct: Use root.after() or queue

    # Mistake 3: Not handling exceptions in callbacks
    def risky_callback():
        try:
            # Some risky operation
            result = int("not_a_number")
        except ValueError as e:
            tk.messagebox.showerror("Error", f"Invalid input: {e}")

    tk.Button(root, text="Risky Operation", command=risky_callback).pack(pady=20)

    # Mistake 4: Using pack/grid/place inconsistently
    # Better to stick with one layout manager per container

    # Mistake 5: Not configuring grid weights
    frame = tk.Frame(root)
    frame.pack(fill=tk.BOTH, expand=True)

    # Without weight, widgets won't expand properly
    tk.Button(frame, text="Button 1").grid(row=0, column=0)
    tk.Button(frame, text="Button 2").grid(row=0, column=1)

    # Configure weights for proper expansion
    frame.grid_columnconfigure(0, weight=1)
    frame.grid_columnconfigure(1, weight=1)

    return root  # Don't call mainloop here!

# Latihan 1: Kalkulator Sederhana
import tkinter as tk

class SimpleCalculator:
    def __init__(self, root):
        self.root = root
        self.root.title("Kalkulator Sederhana")
        self.root.geometry("300x400")

        self.expression = ""

        self.create_widgets()

    def create_widgets(self):
        # Display
        self.display = tk.Entry(self.root, font=("Arial", 20), justify="right")
        self.display.pack(fill=tk.X, padx=10, pady=10)

        # Button frame
        button_frame = tk.Frame(self.root)
        button_frame.pack()

        # Button layout
        buttons = [
            ['7', '8', '9', '/'],
            ['4', '5', '6', '*'],
            ['1', '2', '3', '-'],
            ['0', '.', '=', '+']
        ]

        for row in buttons:
            row_frame = tk.Frame(button_frame)
            row_frame.pack()

            for button_text in row:
                if button_text == '=':
                    tk.Button(row_frame, text=button_text, width=5, height=2,
                             command=self.calculate).pack(side=tk.LEFT, padx=2, pady=2)
                elif button_text in ['/', '*', '-', '+']:
                    tk.Button(row_frame, text=button_text, width=5, height=2,
                             command=lambda op=button_text: self.append_operator(op)).pack(side=tk.LEFT, padx=2, pady=2)
                else:
                    tk.Button(row_frame, text=button_text, width=5, height=2,
                             command=lambda num=button_text: self.append_number(num)).pack(side=tk.LEFT, padx=2, pady=2)

        # Clear button
        tk.Button(self.root, text="Clear", command=self.clear).pack(pady=10)

    def append_number(self, number):
        self.expression += number
        self.display.delete(0, tk.END)
        self.display.insert(0, self.expression)

    def append_operator(self, operator):
        if self.expression and self.expression[-1] not in ['+', '-', '*', '/']:
            self.expression += operator
            self.display.delete(0, tk.END)
            self.display.insert(0, self.expression)

    def calculate(self):
        try:
            result = eval(self.expression)
            self.display.delete(0, tk.END)
            self.display.insert(0, str(result))
            self.expression = str(result)
        except:
            self.display.delete(0, tk.END)
            self.display.insert(0, "Error")
            self.expression = ""

    def clear(self):
        self.expression = ""
        self.display.delete(0, tk.END)

class TodoApp:
    def __init__(self, root):
        self.root = root
        self.root.title("To-Do List")
        self.root.geometry("400x500")

        self.tasks = []
        self.create_widgets()

    def create_widgets(self):
        # Input frame
        input_frame = tk.Frame(self.root)
        input_frame.pack(fill=tk.X, padx=10, pady=10)

        self.task_entry = tk.Entry(input_frame, font=("Arial", 12))
        self.task_entry.pack(side=tk.LEFT, fill=tk.X, expand=True)

        tk.Button(input_frame, text="Add Task", command=self.add_task).pack(side=tk.RIGHT, padx=(5,0))

        # List frame
        list_frame = tk.Frame(self.root)
        list_frame.pack(fill=tk.BOTH, expand=True, padx=10, pady=(0,10))

        # Scrollbar
        scrollbar = tk.Scrollbar(list_frame)
        scrollbar.pack(side=tk.RIGHT, fill=tk.Y)

        self.task_listbox = tk.Listbox(list_frame, font=("Arial", 12),
                                      yscrollcommand=scrollbar.set)
        self.task_listbox.pack(fill=tk.BOTH, expand=True)
        scrollbar.config(command=self.task_listbox.yview)

        # Button frame
        button_frame = tk.Frame(self.root)
        button_frame.pack(fill=tk.X, padx=10, pady=(0,10))

        tk.Button(button_frame, text="Delete Task", command=self.delete_task).pack(side=tk.LEFT)
        tk.Button(button_frame, text="Mark Complete", command=self.mark_complete).pack(side=tk.LEFT, padx=(5,0))
        tk.Button(button_frame, text="Clear All", command=self.clear_all).pack(side=tk.RIGHT)

    def add_task(self):
        task = self.task_entry.get().strip()
        if task:
            self.tasks.append({"text": task, "completed": False})
            self.update_listbox()
            self.task_entry.delete(0, tk.END)

    def delete_task(self):
        selection = self.task_listbox.curselection()
        if selection:
            index = selection[0]
            del self.tasks[index]
            self.update_listbox()

    def mark_complete(self):
        selection = self.task_listbox.curselection()
        if selection:
            index = selection[0]
            self.tasks[index]["completed"] = True
            self.update_listbox()

    def clear_all(self):
        self.tasks.clear()
        self.update_listbox()

    def update_listbox(self):
        self.task_listbox.delete(0, tk.END)
        for task in self.tasks:
            status = "‚úì" if task["completed"] else "‚óã"
            display_text = f"{status} {task['text']}"
            self.task_listbox.insert(tk.END, display_text)

class SimpleTextEditor:
    def __init__(self, root):
        self.root = root
        self.root.title("Simple Text Editor")
        self.root.geometry("600x500")

        self.file_path = None
        self.create_widgets()

    def create_widgets(self):
        # Menu bar
        menubar = tk.Menu(self.root)
        self.root.config(menu=menubar)

        # File menu
        file_menu = tk.Menu(menubar, tearoff=0)
        menubar.add_cascade(label="File", menu=file_menu)
        file_menu.add_command(label="New", command=self.new_file)
        file_menu.add_command(label="Open", command=self.open_file)
        file_menu.add_command(label="Save", command=self.save_file)
        file_menu.add_separator()
        file_menu.add_command(label="Exit", command=self.root.quit)

        # Text area
        self.text_area = tk.Text(self.root, font=("Consolas", 12), wrap=tk.WORD)
        self.text_area.pack(fill=tk.BOTH, expand=True, padx=5, pady=5)

        # Status bar
        self.status_bar = tk.Label(self.root, text="Ready", relief=tk.SUNKEN, anchor="w")
        self.status_bar.pack(fill=tk.X, side=tk.BOTTOM)

        # Bind events
        self.text_area.bind("", self.update_status)

    def new_file(self):
        self.text_area.delete(1.0, tk.END)
        self.file_path = None
        self.root.title("Simple Text Editor - New File")
        self.status_bar.config(text="New file created")

    def open_file(self):
        from tkinter import filedialog
        file_path = filedialog.askopenfilename(filetypes=[("Text files", "*.txt"), ("All files", "*.*")])
        if file_path:
            try:
                with open(file_path, 'r') as file:
                    content = file.read()
                    self.text_area.delete(1.0, tk.END)
                    self.text_area.insert(1.0, content)
                    self.file_path = file_path
                    self.root.title(f"Simple Text Editor - {file_path}")
                    self.status_bar.config(text=f"Opened: {file_path}")
            except Exception as e:
                tk.messagebox.showerror("Error", f"Failed to open file: {e}")

    def save_file(self):
        if self.file_path:
            try:
                content = self.text_area.get(1.0, tk.END)
                with open(self.file_path, 'w') as file:
                    file.write(content)
                self.status_bar.config(text=f"Saved: {self.file_path}")
            except Exception as e:
                tk.messagebox.showerror("Error", f"Failed to save file: {e}")
        else:
            self.save_file_as()

    def save_file_as(self):
        from tkinter import filedialog
        file_path = filedialog.asksaveasfilename(defaultextension=".txt",
                                                filetypes=[("Text files", "*.txt"), ("All files", "*.*")])
        if file_path:
            self.file_path = file_path
            self.save_file()

    def update_status(self, event=None):
        content = self.text_area.get(1.0, tk.END)
        lines = len(content.split('\n'))
        words = len(content.split())
        chars = len(content) - 1  # Subtract 1 for the extra newline
        self.status_bar.config(text=f"Lines: {lines} | Words: {words} | Characters: {chars}")
