3Ô∏è‚É£ Fungsi dengan Return Value

Fungsi yang mengembalikan hasil perhitungan.

def hitung_luas_persegi(sisi):
    """Menghitung luas persegi"""
    luas = sisi * sisi
    return luas

# Menggunakan return value
hasil1 = hitung_luas_persegi(5)
hasil2 = hitung_luas_persegi(10)

print(f"Luas persegi sisi 5: {hasil1}")
print(f"Luas persegi sisi 10: {hasil2}")

# Menggunakan langsung
print(f"Luas persegi sisi 7: {hitung_luas_persegi(7)}")

4Ô∏è‚É£ Fungsi dengan Multiple Parameters

Fungsi yang menerima beberapa parameter.

def hitung_luas_persegi_panjang(panjang, lebar):
    """Menghitung luas persegi panjang"""
    return panjang * lebar

def perkenalan(nama, umur, kota):
    """Memperkenalkan diri"""
    return f"Halo, nama saya {nama}, umur {umur} tahun, tinggal di {kota}."

# Memanggil fungsi
luas = hitung_luas_persegi_panjang(10, 5)
print(f"Luas: {luas}")

intro = perkenalan("John", 25, "Jakarta")
print(intro)

5Ô∏è‚É£ Fungsi dengan Default Parameter

Fungsi dengan parameter yang memiliki nilai default.

def sapa_lengkap(nama, salam="Halo", waktu="pagi"):
    """Menyapa dengan salam dan waktu"""
    return f"{salam}! Selamat {waktu}, {nama}!"

# Memanggil dengan berbagai cara
print(sapa_lengkap("Alice"))                    # Menggunakan semua default
print(sapa_lengkap("Bob", "Selamat pagi"))     # Mengubah salam
print(sapa_lengkap("Charlie", "Hai", "siang")) # Mengubah semua

# Menggunakan keyword arguments
print(sapa_lengkap(nama="David", waktu="malam", salam="Selamat"))

6Ô∏è‚É£ Fungsi dengan *args

Fungsi yang menerima jumlah argument yang tidak terbatas.

def jumlahkan_semua(*angka):
    """Menjumlahkan semua angka yang diberikan"""
    total = 0
    for num in angka:
        total += num
    return total

def cetak_daftar_belanja(*item):
    """Mencetak daftar belanja"""
    print("Daftar Belanja:")
    for i, barang in enumerate(item, 1):
        print(f"{i}. {barang}")

# Memanggil fungsi
print(f"Total: {jumlahkan_semua(1, 2, 3, 4, 5)}")
print(f"Total: {jumlahkan_semua(10, 20, 30)}")

cetak_daftar_belanja("Apel", "Jeruk", "Pisang", "Mangga")

7Ô∏è‚É£ Fungsi dengan **kwargs

Fungsi yang menerima keyword arguments yang tidak terbatas.

def buat_profil(**info):
    """Membuat profil dari informasi yang diberikan"""
    print("=== PROFIL ===")
    for key, value in info.items():
        print(f"{key.capitalize()}: {value}")
    print("=" * 12)

# Memanggil fungsi
buat_profil(nama="Alice", umur=25, pekerjaan="Programmer", kota="Jakarta")
print()
buat_profil(nama="Bob", hobi="Fotografi", pendidikan="S1 Informatika")

8Ô∏è‚É£ Kombinasi Semua Konsep

Fungsi yang menggunakan semua konsep yang telah dipelajari.

def kalkulator(operasi, *angka, **opsi):
    """Kalkulator yang fleksibel"""
    if operasi == "tambah":
        hasil = sum(angka)
    elif operasi == "kali":
        hasil = 1
        for num in angka:
            hasil *= num
    elif operasi == "rata_rata":
        hasil = sum(angka) / len(angka) if angka else 0
    else:
        return "Operasi tidak dikenal"

    # Opsi tambahan
    if opsi.get("tampilkan_proses", False):
        print(f"Angka: {angka}")
        print(f"Operasi: {operasi}")

    if opsi.get("bulatkan", False):
        hasil = round(hasil)

    return hasil

# Menggunakan fungsi
print(kalkulator("tambah", 1, 2, 3, 4, 5))
print(kalkulator("kali", 2, 3, 4))
print(kalkulator("rata_rata", 85, 90, 88, 92))
print(kalkulator("tambah", 1.5, 2.7, 3.2, bulatkan=True, tampilkan_proses=True))
